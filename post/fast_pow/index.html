ll fast_mul(ll a, ll b, ll p)
{
    ll res = 0;
    while (b)
    {
        if (b & 1) res = (res + a) % p;
        a = a * 2 % p;
        b >>= 1;
    }

}
ll fast_pow(ll a, ll b, ll p)
{
    ll res = 1;
    while (b)
    {
        if (b & 1) res = fast_mul(res, a, p);
        a = fast_mul(a, a, p);
        b >>= 1;
    }
    return res;
}
