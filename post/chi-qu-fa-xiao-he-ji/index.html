#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int N = 1e5 + 10;
int n, s,a[N];
int main(){
	int t;
	cin >> t;
	while (t--)
	{
		scanf("%d%d", &n,&s);
		for (int i = 1; i <= n; i++)
			scanf("%d", &a[i]);
		int st = 1, en = 1;
		int tot = 0;
		int ans = n+10;
		while (1)
		{
			while (en <= n&&tot < s)
				tot += a[en++];
			if (tot < s)
				break;
			ans = min(ans, en - st);
			tot -= a[st++];
		}
		if (ans == n + 10) printf("0\n");
		else 
		printf("%d\n", ans);
	}
	return 0;
}
