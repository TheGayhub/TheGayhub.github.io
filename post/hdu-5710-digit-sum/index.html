#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int N = 2e5 + 10;
int a, b;
int gcd(int a, int b)
{
    if (b == 0) return a;
    return gcd(b, a%b);
}
int ans[N];
int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        scanf("%d%d", &a, &b);
        int d = gcd(a, b);
        a /= d, b /= d;
        int l = 2 * b - a, sn = 9 * b;
        if (l<0 || l * 5>sn){
            puts("0");
            continue;
        }
        d = gcd(l, sn);
        l /= d, sn /= d;
        int tot = 0;
        sn -= 5 * l;
        for (int i = 0; i < l; i++)
        {
            int tmp = min(4, sn);
            ans[tot++] = 5 + tmp;
            sn -= tmp;
        }
        while (sn)
        {
            int tmp = min(4, sn);
            ans[tot++] = tmp;
            sn -= tmp;
        }
        for (int i = tot - 1; i >= 0; i--)
            cout << ans[i];
        puts("");
    }
    return 0;
}
